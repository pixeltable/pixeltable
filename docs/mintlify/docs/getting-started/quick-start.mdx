## Basic Chatbot Example

```python
import pixeltable as pxt
from pixeltable.functions import openai

# Create table
conversations = pxt.create_table(
    path_str="chatbot", schema_or_df={"prompt": pxt.String}, if_exists="ignore"
)

# Create message column
conversations.add_computed_column(messages=[{"role": "user", "content": conversations.prompt}])

# Create response column
conversations.add_computed_column(response=openai.chat_completions(
    messages=conversations.messages,
    model="gpt-4o-mini",
    ))

# Create answer column
conversations.add_computed_column(answer=conversations.response.choices[0].message.content)

# # Lets test it!!
conversations.insert(
    [{"prompt": "Write a haiku about dogs."}]
)

# Print the results
print(conversations.collect())
```
